```
  0,

  0, // INSTRUMENT NUM
  0,

  0, // TRIM LEFT
  0,
  0,

  0, // TRIM RIGHT
  0,
  0,
  0,
  0,
  248,
  66,
  100, // VOLUME (0-200)
  0,
  0, // SIGN - PITCH / PAN
  0, // ATACK
  255, // ONE - 255, KEY - 255, LEG - 15,  RELEASE ... looks like multipurpose byte
  0,
  0,
  0, // KEY - 1, LEG - 2,
  60, // PAD ID
  0,
  0 // PITCH B
]
```
